id_nums = pd.read_csv('身份证.csv', engine='python', encoding='utf-8', index_col=0)
id_nums.loc[:, '性别身份证号'] = id_nums.loc[:, '性别身份证号'].apply(lambda s:s[1:])
id_nums = id_nums.rename(columns={'性别身份证号':'身份证号'})
id_nums = id_nums[~id_nums.index.duplicated(keep='first')]
native_place = pd.DataFrame(np.zeros([len(id_nums), len(areas)], dtype=int))
native_place.columns = areas.values()
native_place.index = id_nums.index
native_place = native_place[~native_place.index.duplicated(keep='first')]
def one_hot(row, id_nums):
    name = row.name
    first_2_digits = id_nums.loc[name]['身份证号'][:2]
    area = areas[int(first_2_digits)]
    row[area] = 1
    return row

native_place = native_place.apply(one_hot, axis=1, args=(id_nums, ))

N = len(fitness_test_dorm_numeric)
euclidean_dist = np.zeros([N, N])
manhattan_dist = np.zeros([N, N])
mahalanobis_dist = np.zeros([N, N])
incov_m = np.linalg.inv(np.cov(fitness_test_dorm_numeric.as_matrix().T))
for i in range(N):
    for j in range(N):
        euclidean_dist[i,j] = np.linalg.norm(fitness_test_dorm_numeric.iloc[i] - fitness_test_dorm_numeric.iloc[j])
        manhattan_dist[i,j] = np.linalg.norm(fitness_test_dorm_numeric.iloc[i] - fitness_test_dorm_numeric.iloc[j])
        diff = fitness_test_dorm_numeric.iloc[i] - fitness_test_dorm_numeric.iloc[j]
        mahalanobis_dist[i, j] = np.sqrt(diff.T.dot(incov_m).dot(diff))
